72        #[derive(Debug, Clone, Copy, Default, PartialEq, Eq)]
73        pub struct $name;
74
75        impl ConstString for $name {
76            const VALUE: &str = $value;
77        }
78
79        impl serde::Serialize for $name {
80            fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
81            where
82                S: serde::Serializer,
83            {
84                $value.serialize(serializer)
85            }
86        }
87
88        impl<'de> serde::Deserialize<'de> for $name {
89            fn deserialize<D>(deserializer: D) -> Result<$name, D::Error>
90            where
91                D: serde::Deserializer<'de>,
92            {
93                let s: String = serde::Deserialize::deserialize(deserializer)?;
94                if s == $value {
95                    Ok($name)
96                } else {
97                    Err(serde::de::Error::custom(format!(concat!(
98                        "expect const string value \"",
99                        $value,
100                        "\""
101                    ))))
102                }
103            }
104        }
105
106        #[cfg(feature = "schemars")]
107        impl schemars::JsonSchema for $name {
108            fn schema_name() -> Cow<'static, str> {
109                Cow::Borrowed(stringify!($name))
110            }
111
112            fn json_schema(_: &mut schemars::SchemaGenerator) -> schemars::Schema {
113                use serde_json::{Map, json};
114
115                let mut schema_map = Map::new();
116                schema_map.insert("type".to_string(), json!("string"));
117                schema_map.insert("format".to_string(), json!("const"));
118                schema_map.insert("const".to_string(), json!($value));
119
120                schemars::Schema::from(schema_map)
121            }
122        }
123    };
124}

186/// A flexible identifier type that can be either a number or a string.
187///
188/// This is commonly used for request IDs and other identifiers in JSON-RPC
189/// where the specification allows both numeric and string values.
190#[derive(Debug, Clone, Eq, PartialEq, Hash)]

203        }
204    }
205}

209        match self {
210            NumberOrString::Number(n) => n.fmt(f),
211            NumberOrString::String(s) => s.fmt(f),

217    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
218    where
219        S: serde::Serializer,
220    {
221        match self {
222            NumberOrString::Number(n) => n.serialize(serializer),
223            NumberOrString::String(s) => s.serialize(serializer),
224        }
225    }
226}

228impl<'de> Deserialize<'de> for NumberOrString {
229    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
230    where
231        D: serde::Deserializer<'de>,
232    {
233        let value: Value = Deserialize::deserialize(deserializer)?;
234        match value {
235            Value::Number(n) => {
236                if let Some(i) = n.as_i64() {
237                    Ok(NumberOrString::Number(i))
238                } else if let Some(u) = n.as_u64() {
239                    // Handle large unsigned numbers that fit in i64
240                    if u <= i64::MAX as u64 {
241                        Ok(NumberOrString::Number(u as i64))
242                    } else {
243                        Err(serde::de::Error::custom("Number too large for i64"))
244                    }
245                } else {
246                    Err(serde::de::Error::custom("Expected an integer"))
247                }
248            }
249            Value::String(s) => Ok(NumberOrString::String(s.into())),
250            _ => Err(serde::de::Error::custom("Expect number or string")),
251        }
252    }
253}

515/// Represents any JSON-RPC message that can be sent or received.
516///
517/// This enum covers all possible message types in the JSON-RPC protocol:
518/// individual requests/responses, notifications, and errors.
519/// It serves as the top-level message container for MCP communication.
520#[derive(Debug, Serialize, Deserialize, Clone, PartialEq)]
521#[serde(untagged)]
522#[cfg_attr(feature = "schemars", derive(schemars::JsonSchema))]
523pub enum JsonRpcMessage<Req = Request, Resp = DefaultResponse, Noti = Notification> {
524    /// A single request expecting a response
525    Request(JsonRpcRequest<Req>),
526    /// A response to a previous request
527    Response(JsonRpcResponse<Resp>),
528    /// A one-way notification (no response expected)
529    Notification(JsonRpcNotification<Noti>),
530    /// An error response
531    Error(JsonRpcError),
532}
533
534impl<Req, Resp, Not> JsonRpcMessage<Req, Resp, Not> {
535    #[inline]
536    pub const fn request(request: Req, id: RequestId) -> Self {
537        JsonRpcMessage::Request(JsonRpcRequest {
538            jsonrpc: JsonRpcVersion2_0,
539            id,
540            request,
541        })
542    }
543    #[inline]
544    pub const fn response(response: Resp, id: RequestId) -> Self {
545        JsonRpcMessage::Response(JsonRpcResponse {
546            jsonrpc: JsonRpcVersion2_0,
547            id,
548            result: response,
549        })
550    }
551    #[inline]
552    pub const fn error(error: ErrorData, id: RequestId) -> Self {
553        JsonRpcMessage::Error(JsonRpcError {
554            jsonrpc: JsonRpcVersion2_0,
555            id,
556            error,
557        })
558    }
559    #[inline]
560    pub const fn notification(notification: Not) -> Self {
561        JsonRpcMessage::Notification(JsonRpcNotification {
562            jsonrpc: JsonRpcVersion2_0,
563            notification,
564        })
565    }
566    pub fn into_request(self) -> Option<(Req, RequestId)> {
567        match self {
568            JsonRpcMessage::Request(r) => Some((r.request, r.id)),
569            _ => None,
570        }
571    }
572    pub fn into_response(self) -> Option<(Resp, RequestId)> {
573        match self {
574            JsonRpcMessage::Response(r) => Some((r.result, r.id)),
575            _ => None,
576        }
577    }
578    pub fn into_notification(self) -> Option<Not> {
579        match self {
580            JsonRpcMessage::Notification(n) => Some(n.notification),
581            _ => None,
582        }
583    }
584    pub fn into_error(self) -> Option<(ErrorData, RequestId)> {
585        match self {
586            JsonRpcMessage::Error(e) => Some((e.error, e.id)),
587            _ => None,
588        }
589    }
590    pub fn into_result(self) -> Option<(Result<Resp, ErrorData>, RequestId)> {
591        match self {
592            JsonRpcMessage::Response(r) => Some((Ok(r.result), r.id)),
593            JsonRpcMessage::Error(e) => Some((Err(e.error), e.id)),
594
595            _ => None,
596        }
597    }
598}

700    ///

710            params,

717        self.params
718            .as_ref()
719            .map(|params| serde_json::from_value(params.clone()))

721    }

1592impl CreateMessageRequestParams {
1593    /// Validate the sampling request parameters per SEP-1577 spec requirements.
1594    ///
1595    /// Checks:
1596    /// - ToolUse content is only allowed in assistant messages
1597    /// - ToolResult content is only allowed in user messages
1598    /// - Messages with tool result content MUST NOT contain other content types
1599    /// - Every assistant ToolUse must be balanced with a corresponding user ToolResult
1600    pub fn validate(&self) -> Result<(), String> {
1601        for msg in &self.messages {
1602            for content in msg.content.iter() {
1603                // ToolUse only in assistant messages, ToolResult only in user messages
1604                match content {
1605                    SamplingMessageContent::ToolUse(_) if msg.role != Role::Assistant => {
1606                        return Err("ToolUse content is only allowed in assistant messages".into());
1607                    }
1608                    SamplingMessageContent::ToolResult(_) if msg.role != Role::User => {
1609                        return Err("ToolResult content is only allowed in user messages".into());
1610                    }
1611                    _ => {}
1612                }
1613            }
1614
1615            // Tool result messages MUST NOT contain other content types
1616            let contents: Vec<_> = msg.content.iter().collect();
1617            let has_tool_result = contents
1618                .iter()
1619                .any(|c| matches!(c, SamplingMessageContent::ToolResult(_)));
1620            if has_tool_result
1621                && contents
1622                    .iter()
1623                    .any(|c| !matches!(c, SamplingMessageContent::ToolResult(_)))
1624            {
1625                return Err(
1626                    "SamplingMessage with tool result content MUST NOT contain other content types"
1627                        .into(),
1628                );
1629            }
1630        }
1631
1632        // Every assistant ToolUse must be balanced with a user ToolResult
1633        self.validate_tool_use_result_balance()?;
1634
1635        Ok(())
1636    }
1637
1638    fn validate_tool_use_result_balance(&self) -> Result<(), String> {
1639        let mut pending_tool_use_ids: Vec<String> = Vec::new();
1640        for msg in &self.messages {
1641            if msg.role == Role::Assistant {
1642                for content in msg.content.iter() {
1643                    if let SamplingMessageContent::ToolUse(tu) = content {
1644                        pending_tool_use_ids.push(tu.id.clone());
1645                    }
1646                }
1647            } else if msg.role == Role::User {
1648                for content in msg.content.iter() {
1649                    if let SamplingMessageContent::ToolResult(tr) = content {
1650                        if !pending_tool_use_ids.contains(&tr.tool_use_id) {
1651                            return Err(format!(
1652                                "ToolResult with toolUseId '{}' has no matching ToolUse",
1653                                tr.tool_use_id
1654                            ));
1655                        }
1656                        pending_tool_use_ids.retain(|id| id != &tr.tool_use_id);
1657                    }
1658                }
1659            }
1660        }
1661        if !pending_tool_use_ids.is_empty() {
1662            return Err(format!(
1663                "ToolUse with id(s) {:?} not balanced with ToolResult",
1664                pending_tool_use_ids
1665            ));
1666        }
1667        Ok(())
1668    }
1669}
