8#[derive(FromMeta, Default, Debug)]
9#[darling(default)]
10pub struct PromptAttribute {
11    /// The name of the prompt
12    pub name: Option<String>,
13    /// Human readable title of prompt
14    pub title: Option<String>,
15    /// Optional description of what the prompt does
16    pub description: Option<String>,
17    /// Arguments that can be passed to the prompt
18    pub arguments: Option<Expr>,
19    /// Optional icons for the prompt
20    pub icons: Option<Expr>,
21    /// Optional metadata for the prompt
22    pub meta: Option<Expr>,
23}

25pub struct ResolvedPromptAttribute {
26    pub name: String,
27    pub title: Option<String>,
28    pub description: Option<Expr>,
29    pub arguments: Expr,
30    pub icons: Option<Expr>,
31    pub meta: Option<Expr>,
32}

34impl ResolvedPromptAttribute {
35    pub fn into_fn(self, fn_ident: Ident) -> syn::Result<ImplItemFn> {
36        let Self {
37            name,
38            description,
39            arguments,
40            title,
41            icons,
42            meta,
43        } = self;
44        let description = if let Some(description) = description {
45            quote! { Some(#description.into()) }
46        } else {
47            quote! { None }
48        };
49        let title = if let Some(title) = title {
50            quote! { Some(#title.into()) }
51        } else {
52            quote! { None }
53        };
54        let icons = if let Some(icons) = icons {
55            quote! { Some(#icons) }
56        } else {
57            quote! { None }
58        };
59        let meta = if let Some(meta) = meta {
60            quote! { Some(#meta) }
61        } else {
62            quote! { None }
63        };
64        let tokens = quote! {
65            pub fn #fn_ident() -> rmcp::model::Prompt {
66                rmcp::model::Prompt {
67                    name: #name.into(),
68                    description: #description,
69                    arguments: #arguments,
70                    title: #title,
71                    icons: #icons,
72                    meta: #meta,
73                }
74            }
75        };
76        syn::parse2::<ImplItemFn>(tokens)
77    }
78}

80pub fn prompt(attr: TokenStream, input: TokenStream) -> syn::Result<TokenStream> {
81    let attribute = if attr.is_empty() {
82        Default::default()
83    } else {
84        let attr_args = NestedMeta::parse_meta_list(attr)?;
85        PromptAttribute::from_list(&attr_args)?
86    };
87    let mut fn_item = syn::parse2::<ImplItemFn>(input.clone())?;
88    let fn_ident = &fn_item.sig.ident;
89
90    let prompt_attr_fn_ident = format_ident!("{}_prompt_attr", fn_ident);
91
92    // Try to find prompt parameters from function parameters
93    let arguments_expr = if let Some(arguments) = attribute.arguments {
94        arguments
95    } else {
96        // Look for a type named Parameters in the function signature
97        let params_ty = crate::common::find_parameters_type_impl(&fn_item);
98
99        if let Some(params_ty) = params_ty {
100            // Generate arguments from the type's schema with caching
101            syn::parse2::<Expr>(quote! {
102                rmcp::handler::server::prompt::cached_arguments_from_schema::<#params_ty>()
103            })?
104        } else {
105            // No arguments
106            none_expr()?
107        }
108    };
109
110    let name = attribute.name.unwrap_or_else(|| fn_ident.to_string());
111    let description = if let Some(s) = attribute.description {
112        Some(Expr::Lit(syn::ExprLit {
113            attrs: Vec::new(),
114            lit: syn::Lit::Str(syn::LitStr::new(&s, Span::call_site())),
115        }))
116    } else {
117        fn_item.attrs.iter().try_fold(None, extract_doc_line)?
118    };
119    let arguments = arguments_expr;
120
121    let resolved_prompt_attr = ResolvedPromptAttribute {
122        name: name.clone(),
123        description: description.clone(),
124        arguments: arguments.clone(),
125        title: attribute.title,
126        icons: attribute.icons,
127        meta: attribute.meta,
128    };
129    let prompt_attr_fn = resolved_prompt_attr.into_fn(prompt_attr_fn_ident.clone())?;
130
131    // Modify the input function for async support (same as tool macro)
132    if fn_item.sig.asyncness.is_some() {
133        // 1. remove asyncness from sig
134        // 2. make return type: `futures::future::BoxFuture<'_, #ReturnType>`
135        // 3. make body: { Box::pin(async move { #body }) }
136        let new_output = syn::parse2::<ReturnType>({
137            let mut lt = quote! { 'static };
138            if let Some(receiver) = fn_item.sig.receiver() {
139                if let Some((_, receiver_lt)) = receiver.reference.as_ref() {
140                    if let Some(receiver_lt) = receiver_lt {
141                        lt = quote! { #receiver_lt };
142                    } else {
143                        lt = quote! { '_ };
144                    }
145                }
146            }
147            match &fn_item.sig.output {
148                syn::ReturnType::Default => {
149                    quote! { -> ::std::pin::Pin<Box<dyn ::std::future::Future<Output = ()> + Send + #lt>> }
150                }
151                syn::ReturnType::Type(_, ty) => {
152                    quote! { -> ::std::pin::Pin<Box<dyn ::std::future::Future<Output = #ty> + Send + #lt>> }
153                }
154            }
155        })?;
156        let prev_block = &fn_item.block;
157        let new_block = syn::parse2::<syn::Block>(quote! {
158           { Box::pin(async move #prev_block ) }
159        })?;
160        fn_item.sig.asyncness = None;
161        fn_item.sig.output = new_output;
162        fn_item.block = new_block;
163    }
164
165    Ok(quote! {
166        #prompt_attr_fn
167        #fn_item
168    })
169}

175    #[test]
176    fn test_prompt_macro() -> syn::Result<()> {
177        let attr = quote! {
178            name = "example-prompt",
179            description = "An example prompt"
180        };
181        let input = quote! {
182            async fn example_prompt(&self, Parameters(args): Parameters<ExampleArgs>) -> Result<String> {
183                Ok("Example prompt response".to_string())
184            }
185        };
186        let result = prompt(attr, input)?;
187
188        // Verify the output contains both the attribute function and the modified function
189        let result_str = result.to_string();
190        assert!(result_str.contains("example_prompt_prompt_attr"));
191        assert!(
192            result_str.contains("rmcp")
193                && result_str.contains("model")
194                && result_str.contains("Prompt")
195        );
196
197        Ok(())
198    }
